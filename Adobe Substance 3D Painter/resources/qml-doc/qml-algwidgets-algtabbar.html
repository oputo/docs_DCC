Detailed Description

A tab bar with Substance 3D Painter style. This class is part of the AlgWidgets module.

import AlgWidgets 2.0
import QtQuick 2.7
import QtQuick.Layouts 1.3

ColumnLayout {
  AlgTabBar {
    id: root
    Layout.preferredWidth: 100
    Repeater {
      model: 4
      delegate: AlgTabButton {
        text: index
        onClicked: {
          alg.log.warn(text + " clicked")
        }
      }
    }
  }
}
